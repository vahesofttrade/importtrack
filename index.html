-- ORDER ITEMS table (multiple products per order) create table if not exists public.order_items ( id uuid default gen_random_uuid() primary key, order_id integer references public.orders(id) on delete cascade, product_id uuid references public.products(id) on delete set null, product_name text not null, category text, qty numeric, unit text default 'MT', unit_price numeric, total_price numeric, currency text default 'USD', notes text, created_at timestamptz default now() ); alter table public.order_items enable row level security; drop policy if exists "All users read order_items" on public.order_items; drop policy if exists "Admins manage order_items" on public.order_items; create policy "All users read order_items" on public.order_items for select using (true); create policy "Admins manage order_items" on public.order_items for all using ( exists (select 1 from public.profiles where id = auth.uid() and role = 'admin') ); -- Add total_items count to orders for quick display alter table public.orders add column if not exists items_count integer default 0, add column if not exists has_items boolean default false;
